def post_with_auth(path, payload)
  auth = { client_id: @client_id,
           secret: @secret }
  Plaid::Connect.post(File.join(@env, path), payload.merge(auth))
end

def env_map(env)
  (@env = Plaid.url_from_env(env)) || raise(ArgumentError, 'Invalid value for '                                                                  'Plaid::Client.env '                                                                  "(#{env.inspect}): "                                                                  'must be :sandbox, '                                                                 ':development, or :production ')
end
