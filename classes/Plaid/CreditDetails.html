def get(access_token, account_ids: nil)
  options_payload = {}
  options_payload[:account_ids] = account_ids unless account_ids.nil?
  payload = {
    access_token: access_token,
    options: options_payload
  }

  @client.post_with_auth('credit_details/get', payload)
end
