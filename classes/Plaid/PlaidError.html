def initialize(error_type, error_code, error_message, display_message, request_id)
  @error_type      = error_type
  @error_code      = error_code
  @error_message   = error_message
  @display_message = display_message
  @request_id      = request_id

  super "\n"             "Error Type      : #{@error_type}\n"             "Error Code      : #{@error_code}\n"             "Error Message   : #{@error_message}\n"             "Display Message : #{@display_message}\n"             "Request ID      : #{@request_id}\n"
end
